1112  bool initial_calc = (config->GetExtIter() == 0); // Checks if it is the first calculation.
1113  bool first_iter = (config->GetIntIter() == 0); // Checks if it is the first iteration
1114  bool dynamic = (config->GetDynamic_Analysis() == DYNAMIC); // Dynamic simulations.
1115  bool linear_analysis = (config->GetGeometricConditions() == SMALL_DEFORMATIONS); // Linear analysis.
1116  bool nonlinear_analysis = (config->GetGeometricConditions() == LARGE_DEFORMATIONS); // Nonlinear analysis.
1117  bool newton_raphson = (config->GetKind_SpaceIteScheme_FEA() == NEWTON_RAPHSON); // Newton-Raphson method
1118  bool restart = config->GetRestart(); // Restart analysis
1119  bool initial_calc_restart = (SU2_TYPE::Int(config->GetExtIter()) == config->GetDyn_RestartIter()); // Initial calculation for restart
1120 
1121  bool incremental_load = config->GetIncrementalLoad(); // If an incremental load is applied
1122 
1123  bool body_forces = config->GetDeadLoad(); // Body forces (dead loads).
1124 
1125  /*--- Set vector entries to zero ---*/
1126  for (iPoint = 0; iPoint < geometry->GetnPoint(); iPoint ++) {
1127  LinSysAux.SetBlock_Zero(iPoint);
1128  LinSysRes.SetBlock_Zero(iPoint);
1129  LinSysSol.SetBlock_Zero(iPoint);
1130  }
1131 
1132  /*--- Set matrix entries to zero ---*/
1133 
1134  /*
1135  * If the problem is linear, we only need one Jacobian matrix in the problem, because
1136  * it is going to be constant along the calculations. Therefore, we only initialize
1137  * the Jacobian matrix once, at the beginning of the simulation.
1138  *
1139  * We don't need first_iter, because there is only one iteration per time step in linear analysis.
1140  */
1141  if ((initial_calc && linear_analysis)||
1142  (restart && initial_calc_restart && linear_analysis)){
1143  Jacobian.SetValZero();
1144  }
1145 
1146  /*
1147  * If the problem is dynamic, we need a mass matrix, which will be constant along the calculation
1148  * both for linear and nonlinear analysis. Only initialized once, at the first time step.
1149  *
1150  * The same with the integration constants, as for now we consider the time step to be constant.
1151  *
1152  * We need first_iter, because in nonlinear problems there are more than one subiterations in the first time step.
1153  */
1154  if ((dynamic && initial_calc && first_iter) ||
1155  (dynamic && restart && initial_calc_restart && first_iter)) {
1156  MassMatrix.SetValZero();
1157  Compute_IntegrationConstants(config);
1158  Compute_MassMatrix(geometry, solver_container, numerics, config);
1159  }
1160 
1161  /*
1162  * If body forces are taken into account, we need to compute the term that goes into the residual,
1163  * which will be constant along the calculation both for linear and nonlinear analysis.
1164  *
1165  * Only initialized once, at the first iteration or the beginning of the calculation after a restart.
1166  *
1167  * We need first_iter, because in nonlinear problems there are more than one subiterations in the first time step.
1168  */
1169 
1170  if ((body_forces && initial_calc && first_iter) ||
1171  (body_forces && restart && initial_calc_restart && first_iter)) {
1172  // If the load is incremental, we have to reset the variable to avoid adding up over the increments
1173  if (incremental_load){
1174  for (iPoint = 0; iPoint < nPoint; iPoint++) node[iPoint]->Clear_BodyForces_Res();
1175  }
1176  // Compute the dead load term
1177  Compute_DeadLoad(geometry, solver_container, numerics, config);
1178  }
1179 
1180  /*
1181  * If the problem is nonlinear, we need to initialize the Jacobian and the stiffness matrix at least at the beginning
1182  * of each time step. If the solution method is Newton Rapshon, we initialize it also at the beginning of each
1183  * iteration.
1184  */
1185 
1186  if ((nonlinear_analysis) && ((newton_raphson) || (first_iter))) {
1187  Jacobian.SetValZero();
1188  // StiffMatrix.SetValZero();
1189  }
1190 
1191  /*
1192  * Some external forces may be considered constant over the time step.
1193  */
1194  if (first_iter) {
1195  for (iPoint = 0; iPoint < nPoint; iPoint++) node[iPoint]->Clear_SurfaceLoad_Res();
1196  }
1197 
1198  /*
1199  * If we apply pressure forces, we need to clear the residual on each iteration
1200  */
1201  unsigned short iMarker;
1202  unsigned long iVertex;
1203 
1204  for (iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++)
1205  switch (config->GetMarker_All_KindBC(iMarker)) {
1206  case LOAD_BOUNDARY:
1207  /*--- Only if the load is nonzero - reduces computational cost ---*/
1208  if(config->GetLoad_Value(config->GetMarker_All_TagBound(iMarker)) != 0 ){
1209  /*--- For all the vertices in the marker iMarker ---*/
1210  for (iVertex = 0; iVertex < geometry->nVertex[iMarker]; iVertex++) {
1211  /*--- Retrieve the point ID ---*/
1212  iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
1213  /*--- Clear the residual of the node, to avoid adding on previous values ---*/
1214  node[iPoint]->Clear_SurfaceLoad_Res();
1215  }
1216  }
1217  break;
1218  }
