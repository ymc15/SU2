42 void CSysSolve::GenerateGivens(su2double & dx, su2double & dy, su2double & s, su2double & c) {
43 
44  if ( (dx == 0.0) && (dy == 0.0) ) {
45  c = 1.0;
46  s = 0.0;
47  }
48  else if ( fabs(dy) > fabs(dx) ) {
49  su2double tmp = dx/dy;
50  dx = sqrt(1.0 + tmp*tmp);
51  s = Sign(1.0/dx, dy);

62  dx = 0.0;
63  dy = 0.0;
64  c = 1.0;

66  }

672  MaxIter = config->GetLinear_Solver_Iter() - IterLinSol;
673  IterLinSol += FGMRES_LinSolver(LinSysRes, LinSysSol, *mat_vec, *precond, SolverTol, MaxIter, &Residual, false);

676  }
677  break;
678  }
679 
680  /*--- Dealocate memory of the Krylov subspace method ---*/
681 
682  delete mat_vec;
683  delete precond;
684 
685  }
686 
687  /*--- Smooth the linear system. ---*/
688

691  case SMOOTHER_LUSGS:

693  IterLinSol = Jacobian.LU_SGS_Smoother(LinSysRes, LinSysSol, *mat_vec, SolverTol, MaxIter, &Residual, false, geometry, config);
694  delete mat_vec;
695  break;
696  case SMOOTHER_JACOBI:
697  mat_vec = new CSysMatrixVectorProduct(Jacobian, geometry, config);
698  Jacobian.BuildJacobiPreconditioner();
699  IterLinSol = Jacobian.Jacobi_Smoother(LinSysRes, LinSysSol, *mat_vec, SolverTol, MaxIter, &Residual, false, geometry, config);
700  delete mat_vec;
701  break;
702  case SMOOTHER_ILU:
703  mat_vec = new CSysMatrixVectorProduct(Jacobian, geometry, config);
704  Jacobian.BuildILUPreconditioner();
705  IterLinSol = Jacobian.ILU0_Smoother(LinSysRes, LinSysSol, *mat_vec, SolverTol, MaxIter, &Residual, false, geometry, config);
706  delete mat_vec;
707  break;
708  case SMOOTHER_LINELET:
709  Jacobian.BuildJacobiPreconditioner();
710  Jacobian.ComputeLineletPreconditioner(LinSysRes, LinSysSol, geometry, config);
711  IterLinSol = 1;
712  break;
713  }
714  }
715 
716 
717  if(TapeActive){
718  /*--- Start recording if it was stopped for the linear solver ---*/
719 
720  AD::StartRecording();
721 
722  /*--- Prepare the externally differentiated linear solver ---*/
723 
724  SetExternalSolve(Jacobian, LinSysRes, LinSysSol, geometry, config);
725 
726  }
727 
728  return IterLinSol;
729
