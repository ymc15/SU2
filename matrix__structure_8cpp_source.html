608  }

612  rhs[nVar-1] = rhs[nVar-1] / block[nVar*nVar-1];

616  aux += block[iVar*nVar+jVar]*rhs[jVar];
617  rhs[iVar] = (rhs[iVar]-aux) / block[iVar*nVar+iVar];

1103  int rank = MASTER_NODE;
1104 
1105 #ifdef HAVE_MPI
1106  MPI_Comm_rank(MPI_COMM_WORLD, &rank);

1110 
1111  if (m < 1) {

1113 #ifndef HAVE_MPI

1117  MPI_Finalize();
1118 #endif
1119  }

1123  stored in order to perform multiple iterations of the smoother. ---*/

1126  CSysVector A_x(b);

1141

1145

1147  if ((monitoring) && (rank == MASTER_NODE)) {

1151  cout << " " << i << " " << norm_r/norm0 << endl;
1152  }

1155 
1156  for (i = 0; i < (int)m; i++) {
1157 
1158  /*--- Apply the Jacobi smoother, i.e., multiply by the inverse of the
1159  diagonal matrix of A, which was built in the preprocessing phase. Note
1160  that we are directly updating the solution (x^k+1) during the loop. ---*/
1161

1163  for (iVar = 0; iVar < nVar; iVar++) {

1175

1182

1193  }
