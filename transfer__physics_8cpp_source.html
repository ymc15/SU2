47  CGeometry *flow_geometry, CGeometry *struct_geometry,
48  CConfig *flow_config, CConfig *struct_config){
49 
50  unsigned short iVar;
51 
52  /*--- We have to clear the traction before applying it, because we are "adding" to node and not "setting" ---*/
53 
54  for (unsigned long iPoint = 0; iPoint < struct_geometry->GetnPoint(); iPoint++){
55  struct_solution->node[iPoint]->Clear_FlowTraction();
56  }
57 
58  /*--- Redimensionalize the pressure ---*/
59 
60  su2double *Velocity_ND, *Velocity_Real;
61  su2double Density_ND, Density_Real, Velocity2_Real, Velocity2_ND;
62 
63  Velocity_Real = flow_config->GetVelocity_FreeStream();
64  Density_Real = flow_config->GetDensity_FreeStream();
65 
66  Velocity_ND = flow_config->GetVelocity_FreeStreamND();
67  Density_ND = flow_config->GetDensity_FreeStreamND();
68 
69  Velocity2_Real = 0.0;
70  Velocity2_ND = 0.0;
71  for (iVar = 0; iVar < nVar; iVar++){
72  Velocity2_Real += Velocity_Real[iVar]*Velocity_Real[iVar];
73  Velocity2_ND += Velocity_ND[iVar]*Velocity_ND[iVar];
74  }
75 
76  Physical_Constants[0] = Density_Real*Velocity2_Real/(Density_ND*Velocity2_ND);
77 
78  /*--- Apply a ramp to the transfer of the fluid loads ---*/
79 
80  su2double ModAmpl;
81  su2double CurrentTime = struct_config->GetCurrent_DynTime();
82  su2double Static_Time = struct_config->GetStatic_Time();
83 
84  bool Ramp_Load = struct_config->GetRamp_Load();
85  su2double Ramp_Time = struct_config->GetRamp_Time();
86 
87  bool Sigmoid_Load = struct_config->GetSigmoid_Load();
88  su2double Sigmoid_Time = struct_config->GetSigmoid_Time();
89  su2double Sigmoid_K = struct_config->GetSigmoid_K();
90  su2double SigAux = 0.0;
91 
92  if (CurrentTime <= Static_Time){ ModAmpl=0.0; }
93  else if((CurrentTime > Static_Time) &&
94  (CurrentTime <= (Static_Time + Ramp_Time)) &&
95  (Ramp_Load)){
96  ModAmpl = (CurrentTime-Static_Time) / Ramp_Time;
97  ModAmpl = max(ModAmpl,0.0);
98  ModAmpl = min(ModAmpl,1.0);
99  Physical_Constants[1] = ModAmpl;
100  }
101  else if((CurrentTime > Static_Time) &&
102  (CurrentTime <= (Static_Time + Sigmoid_Time)) &&
103  (Sigmoid_Load)){
104  SigAux = (CurrentTime-Static_Time) / Sigmoid_Time;
105  ModAmpl = (1 / (1+exp(-1*Sigmoid_K*(SigAux - 0.5)) ) );
106  ModAmpl = max(ModAmpl,0.0);
107  ModAmpl = min(ModAmpl,1.0);
108  Physical_Constants[1] = ModAmpl;
109  }
110  else{ Physical_Constants[1] = 1.0; }

114 void CTransfer_FlowTraction::GetDonor_Variable(CSolver *flow_solution, CGeometry *flow_geometry, CConfig *flow_config,
115  unsigned long Marker_Flow, unsigned long Vertex_Flow, unsigned long Point_Struct){
116 
117 
118  unsigned short iVar, jVar;
119  unsigned long Point_Flow;
120  su2double *Normal_Flow;
121 
122  // Check the kind of fluid problem
123  bool compressible = (flow_config->GetKind_Regime() == COMPRESSIBLE);
124  bool incompressible = (flow_config->GetKind_Regime() == INCOMPRESSIBLE);
125  bool viscous_flow = ((flow_config->GetKind_Solver() == NAVIER_STOKES) ||
126  (flow_config->GetKind_Solver() == RANS) );
127 
128  // Parameters for the calculations
129  // Pn: Pressure
130  // Pinf: Pressure_infinite
131  // div_vel: Velocity divergence
132  // Dij: Dirac delta
133  su2double Pn = 0.0, div_vel = 0.0, Dij = 0.0;
134  su2double Viscosity = 0.0;
135  su2double **Grad_PrimVar = NULL;
136  su2double Tau[3][3] = { {0.0, 0.0, 0.0} ,
137  {0.0, 0.0, 0.0} ,
138  {0.0, 0.0, 0.0} } ;
139 
140  su2double Pinf = flow_solution->GetPressure_Inf();
141 
142  Point_Flow = flow_geometry->vertex[Marker_Flow][Vertex_Flow]->GetNode();
143  // Get the normal at the vertex: this normal goes inside the fluid domain.
144  Normal_Flow = flow_geometry->vertex[Marker_Flow][Vertex_Flow]->GetNormal();
145 
146  // Retrieve the values of pressure and viscosity
147  if (incompressible){
148 
149  Pn = flow_solution->node[Point_Flow]->GetPressureInc();
150 
151  if (viscous_flow){
152 
153  Grad_PrimVar = flow_solution->node[Point_Flow]->GetGradient_Primitive();
154  Viscosity = flow_solution->node[Point_Flow]->GetLaminarViscosityInc();
155 
156  }
157  }
158  else if (compressible){
159 
160  Pn = flow_solution->node[Point_Flow]->GetPressure();
161 
162  if (viscous_flow){
163 
164  Grad_PrimVar = flow_solution->node[Point_Flow]->GetGradient_Primitive();
165  Viscosity = flow_solution->node[Point_Flow]->GetLaminarViscosity();
166 
167  }
168  }
169 
170  // Calculate tn in the fluid nodes for the inviscid term --> Units of force (non-dimensional).
171  for (iVar = 0; iVar < nVar; iVar++) {
172  Donor_Variable[iVar] = -(Pn-Pinf)*Normal_Flow[iVar];
173  }
174 
175  // Calculate tn in the fluid nodes for the viscous term
176 
177  if ((incompressible || compressible) && viscous_flow){
178 
179  // Divergence of the velocity
180  div_vel = 0.0; for (iVar = 0; iVar < nVar; iVar++) div_vel += Grad_PrimVar[iVar+1][iVar];
181  if (incompressible) div_vel = 0.0;
182 
183  for (iVar = 0; iVar < nVar; iVar++) {
184 
185  for (jVar = 0 ; jVar < nVar; jVar++) {
186  // Dirac delta
187  Dij = 0.0; if (iVar == jVar) Dij = 1.0;
188 
189  // Viscous stress
190  Tau[iVar][jVar] = Viscosity*(Grad_PrimVar[jVar+1][iVar] + Grad_PrimVar[iVar+1][jVar]) -
191  TWO3*Viscosity*div_vel*Dij;
192 
193  // Viscous component in the tn vector --> Units of force (non-dimensional).
194  Donor_Variable[iVar] += Tau[iVar][jVar]*Normal_Flow[jVar];
195  }
196  }
197  }
198 
199  // Redimensionalize and take into account ramp transfer of the loads
200  for (iVar = 0; iVar < nVar; iVar++){
201  Donor_Variable[iVar] = Donor_Variable[iVar] * Physical_Constants[0] * Physical_Constants[1];
202  }
203
