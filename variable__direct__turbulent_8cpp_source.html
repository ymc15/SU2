175 void CTurbSSTVariable::SetBlendingFunc(su2double val_viscosity, su2double val_dist, su2double val_density) {
176  unsigned short iDim;
177  su2double arg2, arg2A, arg2B, arg1;
178 
179  /*--- Cross diffusion ---*/
180 
181  CDkw = 0.0;
182  for (iDim = 0; iDim < nDim; iDim++)
183  CDkw += Gradient[0][iDim]*Gradient[1][iDim];
184  CDkw *= 2.0*val_density*sigma_om2/Solution[1];
185  CDkw = max(CDkw, pow(10.0, -20.0));
186 
187  /*--- F1 ---*/
188 
189  arg2A = sqrt(Solution[0])/(beta_star*Solution[1]*val_dist+EPS*EPS);
190  arg2B = 500.0*val_viscosity / (val_density*val_dist*val_dist*Solution[1]+EPS*EPS);
191  arg2 = max(arg2A, arg2B);
192  arg1 = min(arg2, 4.0*val_density*sigma_om2*Solution[0] / (CDkw*val_dist*val_dist+EPS*EPS));
193  F1 = tanh(pow(arg1, 4.0));
194 
195  /*--- F2 ---*/
196 
197  arg2 = max(2.0*arg2A, arg2B);
198  F2 = tanh(pow(arg2, 2.0));
199 
200 }
