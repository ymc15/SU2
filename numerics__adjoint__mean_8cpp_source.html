478  val_residual_j[iVar] += Proj_ModJac_Tensor[jVar][iVar]*(Psi_i[jVar] - Psi_j[jVar]);
479  }
480 
481  /*--- Implicit contributions, Transpose the matrices and store the Jacobians. Note the negative
482  sign for the ji and jj Jacobians bc the normal direction is flipped. ---*/
483 
484  if (implicit) {
485  for (iVar = 0; iVar < nVar; iVar++) {
486  for (jVar = 0; jVar < nVar; jVar++) {
487  val_Jacobian_ii[jVar][iVar] = Proj_Jac_Tensor_i[iVar][jVar] - Proj_ModJac_Tensor[iVar][jVar];
488  val_Jacobian_ij[jVar][iVar] = Proj_Jac_Tensor_i[iVar][jVar] + Proj_ModJac_Tensor[iVar][jVar];
489  val_Jacobian_ji[jVar][iVar] = -(Proj_Jac_Tensor_j[iVar][jVar] - Proj_ModJac_Tensor[iVar][jVar]);
490  val_Jacobian_jj[jVar][iVar] = -(Proj_Jac_Tensor_j[iVar][jVar] + Proj_ModJac_Tensor[iVar][jVar]);
491  }
492  }
493  }

523 void CCentJST_AdjFlow::ComputeResidual (su2double *val_resconv_i, su2double *val_resvisc_i, su2double *val_resconv_j, su2double *val_resvisc_j,
524  su2double **val_Jacobian_ii, su2double **val_Jacobian_ij, su2double **val_Jacobian_ji, su2double **val_Jacobian_jj,
525  CConfig *config) {
526 
527  /*--- Mean Values ---*/
528 
529  MeanPsiRho = 0.5*(Psi_i[0]+Psi_j[0]);
530  for (iDim = 0; iDim < nDim; iDim++)
531  MeanPhi[iDim] = 0.5*(Psi_i[iDim+1]+Psi_j[iDim+1]);
532  MeanPsiE = 0.5*(Psi_i[nVar-1]+Psi_j[nVar-1]);
533 
534  /*--- Point i convective residual evaluation ---*/
535 
536  ProjVelocity_i = 0; ProjPhi = 0; ProjPhi_Vel = 0; sq_vel = 0; Area = 0;
537  for (iDim = 0; iDim < nDim; iDim++) {
538  Velocity_i[iDim] = U_i[iDim+1] / U_i[0];
539  ProjVelocity_i += Velocity_i[iDim]*Normal[iDim];
540  ProjPhi += MeanPhi[iDim]*Normal[iDim];
541  ProjPhi_Vel += MeanPhi[iDim]*Velocity_i[iDim];
542  sq_vel += 0.5*Velocity_i[iDim]*Velocity_i[iDim];
543  Area += Normal[iDim]*Normal[iDim];
544  }
545  Area = sqrt(Area);
546  phis1 = ProjPhi + ProjVelocity_i*MeanPsiE;
547  phis2 = MeanPsiRho + ProjPhi_Vel + Enthalpy_i*MeanPsiE;
548 
549  val_resconv_i[0] = ProjVelocity_i*MeanPsiRho - phis2*ProjVelocity_i + Gamma_Minus_One*phis1*sq_vel;
550  for (iDim = 0; iDim < nDim; iDim++)
551  val_resconv_i[iDim+1] = ProjVelocity_i*MeanPhi[iDim] + phis2*Normal[iDim] - Gamma_Minus_One*phis1*Velocity_i[iDim];
552  val_resconv_i[nVar-1] = ProjVelocity_i*MeanPsiE + Gamma_Minus_One*phis1;
553 
554  /*--- Flux contributions due to grid movement at point i ---*/
555 
556  if (grid_movement) {
557  su2double ProjGridVel = 0.0;
558  for (iDim = 0; iDim < nDim; iDim++)
559  ProjGridVel += 0.5*(GridVel_i[iDim]+GridVel_j[iDim])*Normal[iDim];
560  val_resconv_i[0] -= ProjGridVel*MeanPsiRho;
561  for (iDim = 0; iDim < nDim; iDim++)
562  val_resconv_i[iDim+1] -= ProjGridVel*MeanPhi[iDim];
563  val_resconv_i[nVar-1] -= ProjGridVel*MeanPsiE;
564  }
565 
566  /*--- Jacobians of the inviscid flux ---*/
567 
568  if (implicit) {
569  val_Jacobian_ii[0][0] = 0.0;
570  for (jDim = 0; jDim < nDim; jDim++)
571  val_Jacobian_ii[0][jDim+1] = -0.5*ProjVelocity_i*Velocity_i[jDim] + Gamma_Minus_One*sq_vel*0.5*Normal[jDim];
572  val_Jacobian_ii[0][nVar-1] = 0.5*ProjVelocity_i*(Gamma_Minus_One*sq_vel - Enthalpy_i);
573  for (iDim = 0; iDim < nDim; iDim++) {
574  val_Jacobian_ii[iDim+1][0] = 0.5*Normal[iDim];
575  for (jDim = 0; jDim < nDim; jDim++)
576  val_Jacobian_ii[iDim+1][jDim+1] = 0.5*Normal[iDim]*Velocity_i[jDim] - 0.5*Gamma_Minus_One*Velocity_i[iDim]*Normal[jDim];
577  val_Jacobian_ii[iDim+1][iDim+1] += 0.5*ProjVelocity_i;
578  val_Jacobian_ii[iDim+1][nVar-1] = 0.5*Enthalpy_i*Normal[iDim] - 0.5*Gamma_Minus_One*Velocity_i[iDim]*ProjVelocity_i;
579  }
580  val_Jacobian_ii[nVar-1][0] = 0;
581  for (jDim = 0; jDim < nDim; jDim++)
582  val_Jacobian_ii[nVar-1][jDim+1] = 0.5*Gamma_Minus_One*Normal[jDim];
583  val_Jacobian_ii[nVar-1][nVar-1] = 0.5*Gamma*ProjVelocity_i;
584 
585  for (iVar = 0; iVar < nVar; iVar++)
586  for (jVar = 0; jVar < nVar; jVar++)
587  val_Jacobian_ij[iVar][jVar] = val_Jacobian_ii[iVar][jVar];
588 
589  /*--- Jacobian contributions due to grid movement at point i ---*/
590 
591  if (grid_movement) {
592  su2double ProjGridVel = 0.0;
593  for (iDim = 0; iDim < nDim; iDim++)
594  ProjGridVel += 0.5*(GridVel_i[iDim]+GridVel_j[iDim])*Normal[iDim];
595  for (iVar = 0; iVar < nVar; iVar++) {
596  val_Jacobian_ii[iVar][iVar] -= 0.5*ProjGridVel;
597  val_Jacobian_ij[iVar][iVar] -= 0.5*ProjGridVel;
598  }
599  }
600  }
601 
602 
603  /*--- Point j convective residual evaluation ---*/
604 
605  ProjVelocity_j = 0; ProjPhi_Vel = 0; sq_vel = 0;
606  for (iDim = 0; iDim < nDim; iDim++) {
607  Velocity_j[iDim] = U_j[iDim+1] / U_j[0];
608  ProjVelocity_j += Velocity_j[iDim]*Normal[iDim];
609  ProjPhi_Vel += MeanPhi[iDim]*Velocity_j[iDim];
610  sq_vel += 0.5*Velocity_j[iDim]*Velocity_j[iDim];
611  }
612 
613  phis1 = ProjPhi + ProjVelocity_j*MeanPsiE;
614  phis2 = MeanPsiRho + ProjPhi_Vel + Enthalpy_j*MeanPsiE;
615 
616  val_resconv_j[0] = -(ProjVelocity_j*MeanPsiRho - phis2*ProjVelocity_j + Gamma_Minus_One*phis1*sq_vel);
617  for (iDim = 0; iDim < nDim; iDim++)
618  val_resconv_j[iDim+1] = -(ProjVelocity_j*MeanPhi[iDim] + phis2*Normal[iDim] - Gamma_Minus_One*phis1*Velocity_j[iDim]);
619  val_resconv_j[nVar-1] = -(ProjVelocity_j*MeanPsiE + Gamma_Minus_One*phis1);
620 
621  /*--- Flux contributions due to grid motion at point j ---*/
622 
623  if (grid_movement) {
624  su2double ProjGridVel = 0.0;
625  for (iDim = 0; iDim < nDim; iDim++)
626  ProjGridVel += 0.5*(GridVel_i[iDim]+GridVel_j[iDim])*Normal[iDim];
627  val_resconv_j[0] += ProjGridVel*MeanPsiRho;
628  for (iDim = 0; iDim < nDim; iDim++)
629  val_resconv_j[iDim+1] += ProjGridVel*MeanPhi[iDim];
630  val_resconv_j[nVar-1] += ProjGridVel*MeanPsiE;
631  }
632 
633  /*--- Jacobians of the inviscid flux ---*/
634 
635  if (implicit) {
636  val_Jacobian_jj[0][0] = 0.0;
637  for (jDim = 0; jDim < nDim; jDim++)
638  val_Jacobian_jj[0][jDim+1] = 0.5*ProjVelocity_j*Velocity_j[jDim] - Gamma_Minus_One*sq_vel*0.5*Normal[jDim];
639  val_Jacobian_jj[0][nVar-1] = -0.5*ProjVelocity_j*(Gamma_Minus_One*sq_vel - Enthalpy_j);
640  for (iDim = 0; iDim < nDim; iDim++) {
641  val_Jacobian_jj[iDim+1][0] = -0.5*Normal[iDim];
642  for (jDim = 0; jDim < nDim; jDim++)
643  val_Jacobian_jj[iDim+1][jDim+1] = -0.5*Normal[iDim]*Velocity_j[jDim] + 0.5*Gamma_Minus_One*Velocity_j[iDim]*Normal[jDim];
644  val_Jacobian_jj[iDim+1][iDim+1] -= 0.5*ProjVelocity_j;
645  val_Jacobian_jj[iDim+1][nVar-1] = -0.5*Enthalpy_j*Normal[iDim] + 0.5*Gamma_Minus_One*Velocity_j[iDim]*ProjVelocity_j;
646  }
647  val_Jacobian_jj[nVar-1][0] = 0;
648  for (jDim = 0; jDim < nDim; jDim++)
649  val_Jacobian_jj[nVar-1][jDim+1] = -0.5*Gamma_Minus_One*Normal[jDim];
650  val_Jacobian_jj[nVar-1][nVar-1] = -0.5*Gamma*ProjVelocity_j;
651 
652  for (iVar = 0; iVar < nVar; iVar++)
653  for (jVar = 0; jVar < nVar; jVar++)
654  val_Jacobian_ji[iVar][jVar] = val_Jacobian_jj[iVar][jVar];
655 
656  /*--- Jacobian contributions due to grid motion at point j ---*/
657 
658  if (grid_movement) {
659  su2double ProjGridVel = 0.0;
660  for (iDim = 0; iDim < nDim; iDim++)
661  ProjGridVel += 0.5*(GridVel_i[iDim]+GridVel_j[iDim])*Normal[iDim];
662  for (iVar = 0; iVar < nVar; iVar++) {
663  val_Jacobian_jj[iVar][iVar] += 0.5*ProjGridVel;
664  val_Jacobian_ji[iVar][iVar] += 0.5*ProjGridVel;
665  }
666  }
667  }
668 
669  /*--- Computes differences btw. variables and Laplacians ---*/
670 
671  for (iVar = 0; iVar < nVar; iVar++) {
672  Diff_Lapl[iVar] = Und_Lapl_i[iVar]-Und_Lapl_j[iVar];
673  Diff_Psi[iVar] = Psi_i[iVar]-Psi_j[iVar];
674  }
675 
676  /*--- Adjustment to projected velocity due to grid motion ---*/
677 
678  if (grid_movement) {
679  su2double ProjGridVel = 0.0;
680  for (iDim = 0; iDim < nDim; iDim++) {
681  ProjGridVel += 0.5*(GridVel_i[iDim]+GridVel_j[iDim])*Normal[iDim];
682  }
683  ProjVelocity_i -= ProjGridVel;
684  ProjVelocity_j += ProjGridVel;
685  }
686 
687  /*--- Compute the spectral radius and stretching factor ---*/
688 
689  Local_Lambda_i = (fabs(ProjVelocity_i)+SoundSpeed_i*Area);
690  Local_Lambda_j = (fabs(ProjVelocity_j)+SoundSpeed_j*Area);
691  MeanLambda = 0.5*(Local_Lambda_i+Local_Lambda_j);
692 
693  Phi_i = pow(Lambda_i/(4.0*MeanLambda), Param_p);
694  Phi_j = pow(Lambda_j/(4.0*MeanLambda), Param_p);
695  StretchingFactor = 4.0*Phi_i*Phi_j/(Phi_i+Phi_j);
696 
697  su2double sc2 = 3.0*(su2double(Neighbor_i)+su2double(Neighbor_j))/(su2double(Neighbor_i)*su2double(Neighbor_j));
698  sc4 = sc2*sc2/4.0;
699  Epsilon_2 = Param_Kappa_2*0.5*(Sensor_i+Sensor_j)*sc2;
700  Epsilon_4 = max(0.0, Param_Kappa_4-Epsilon_2)*sc4;
701 
702  /*--- Compute viscous residual 1st- & 3rd-order dissipation ---*/
703 
704  for (iVar = 0; iVar < nVar; iVar++) {
705  Residual = (Epsilon_2*Diff_Psi[iVar]-Epsilon_4*Diff_Lapl[iVar])*StretchingFactor*MeanLambda;
706  val_resvisc_i[iVar] = -Residual;
707  val_resvisc_j[iVar] = Residual;
708  if (implicit) {
709  val_Jacobian_ii[iVar][iVar] -= Epsilon_2 + su2double(Neighbor_i+1)*Epsilon_4*StretchingFactor*MeanLambda;
710  val_Jacobian_ij[iVar][iVar] += Epsilon_2 + su2double(Neighbor_j+1)*Epsilon_4*StretchingFactor*MeanLambda;
711  val_Jacobian_ji[iVar][iVar] += Epsilon_2 + su2double(Neighbor_i+1)*Epsilon_4*StretchingFactor*MeanLambda;
712  val_Jacobian_jj[iVar][iVar] -= Epsilon_2 + su2double(Neighbor_j+1)*Epsilon_4*StretchingFactor*MeanLambda;
713  }
714  }
715

759  Velocity_i[iDim] = U_i[iDim+1]/DensityInc_i;
760  Velocity_j[iDim] = U_j[iDim+1]/DensityInc_j;

772  }
773  }

776  if (implicit) {
777  for (iVar = 0; iVar < nVar; iVar++)
778  for (jVar = 0; jVar < nVar; jVar++) {

780  val_Jacobian_ij[iVar][jVar] = Proj_Jac_Tensor_i[jVar][iVar];

796  ProjVelocity_j += Velocity_j[iDim]*Normal[iDim];

822  Residual = (Epsilon_2*Diff_Psi[iVar]-Epsilon_4*Diff_Lapl[iVar])*StretchingFactor*MeanLambda;
823  val_resvisc_i[iVar] = -Residual;
824  val_resvisc_j[iVar] = Residual;
825 
826  if (implicit) {
827  val_Jacobian_ii[iVar][iVar] -= Epsilon_2 + Epsilon_4*su2double(Neighbor_i+1)*StretchingFactor*MeanLambda;
828  val_Jacobian_ij[iVar][iVar] += Epsilon_2 + Epsilon_4*su2double(Neighbor_j+1)*StretchingFactor*MeanLambda;
829  val_Jacobian_ji[iVar][iVar] += Epsilon_2 + Epsilon_4*su2double(Neighbor_i+1)*StretchingFactor*MeanLambda;
830  val_Jacobian_jj[iVar][iVar] -= Epsilon_2 + Epsilon_4*su2double(Neighbor_j+1)*StretchingFactor*MeanLambda;
831  }
832  }
