800 void CAdjTurbSolver::ImplicitEuler_Iteration(CGeometry *geometry, CSolver **solver_container, CConfig *config) {
801 
802  unsigned short iVar;
803  unsigned long iPoint, total_index;
804  su2double Delta, Vol;
805 
806  /*--- Set maximum residual to zero ---*/
807 
808  for (iVar = 0; iVar < nVar; iVar++) {
809  SetRes_RMS(iVar, 0.0);
810  SetRes_Max(iVar, 0.0, 0);
811  }
812 
813  /*--- Build implicit system ---*/
814 
815  for (iPoint = 0; iPoint < nPoint; iPoint++) {
816 
817  /*--- Read the volume ---*/
818 
819  Vol = geometry->node[iPoint]->GetVolume();
820 
821  /*--- Modify matrix diagonal to assure diagonal dominance ---*/
822 
823  Delta = Vol / (config->GetCFLRedCoeff_AdjTurb()*solver_container[FLOW_SOL]->node[iPoint]->GetDelta_Time());
824 
825  Jacobian.AddVal2Diag(iPoint, Delta);
826 
827  /*--- Right hand side of the system (-Residual) and initial guess (x = 0) ---*/
828 
829  for (iVar = 0; iVar < nVar; iVar++) {
830  total_index = iPoint*nVar+iVar;
831  LinSysRes[total_index] = -LinSysRes[total_index];
832  LinSysSol[total_index] = 0.0;
833  AddRes_RMS(iVar, LinSysRes[total_index]*LinSysRes[total_index]);
834  AddRes_Max(iVar, fabs(LinSysRes[total_index]), geometry->node[iPoint]->GetGlobalIndex(), geometry->node[iPoint]->GetCoord());
835  }
836 
837  }
838 
839  /*--- Initialize residual and solution at the ghost points ---*/
840 
841  for (iPoint = nPointDomain; iPoint < nPoint; iPoint++) {
842  for (iVar = 0; iVar < nVar; iVar++) {
843  total_index = iPoint*nVar + iVar;
844  LinSysRes[total_index] = 0.0;
845  LinSysSol[total_index] = 0.0;
846  }
847  }
848 
849  /*--- Solve or smooth the linear system ---*/
850 
851  CSysSolve system;
852  system.Solve(Jacobian, LinSysRes, LinSysSol, geometry, config);
853 
854  /*--- Update solution (system written in terms of increments) ---*/
855 
856  for (iPoint = 0; iPoint < nPointDomain; iPoint++) {
857  for (iVar = 0; iVar < nVar; iVar++)
858  node[iPoint]->AddSolution(iVar, LinSysSol[iPoint*nVar+iVar]);
859  }
860 
861  /*--- MPI solution ---*/
862 
863  Set_MPI_Solution(geometry, config);
864 
865  /*--- Compute the root mean square residual ---*/
866 
867  SetResidual_RMS(geometry, config);
868 
869 }
