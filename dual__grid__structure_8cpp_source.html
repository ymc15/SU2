420

423  AD::SetPreaccIn(val_coord_Elem_CG, nDim);

428  vec_a[iDim] = val_coord_Edge_CG[iDim]-val_coord_Point[iDim];

430  vec_c[iDim] = val_coord_Elem_CG[iDim]-val_coord_Point[iDim];

435  vec_d[2] = vec_a[0]*vec_b[1]-vec_a[1]*vec_b[0];
436

440  AD::EndPreacc();
