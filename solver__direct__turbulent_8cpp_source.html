605 void CTurbSolver::ImplicitEuler_Iteration(CGeometry *geometry, CSolver **solver_container, CConfig *config) {
606 
607  unsigned short iVar;
608  unsigned long iPoint, total_index;
609  su2double Delta, Vol, density_old = 0.0, density = 0.0;
610 
611  bool adjoint = config->GetContinuous_Adjoint();
612  bool compressible = (config->GetKind_Regime() == COMPRESSIBLE);
613  bool incompressible = (config->GetKind_Regime() == INCOMPRESSIBLE);
614  bool freesurface = (config->GetKind_Regime() == FREESURFACE);
615 
616  /*--- Set maximum residual to zero ---*/
617 
618  for (iVar = 0; iVar < nVar; iVar++) {
619  SetRes_RMS(iVar, 0.0);
620  SetRes_Max(iVar, 0.0, 0);
621  }
622 
623  /*--- Build implicit system ---*/
624 
625  for (iPoint = 0; iPoint < nPointDomain; iPoint++) {
626 
627  /*--- Read the volume ---*/
628 
629  Vol = geometry->node[iPoint]->GetVolume();
630 
631  /*--- Modify matrix diagonal to assure diagonal dominance ---*/
632 
633  Delta = Vol / (config->GetCFLRedCoeff_Turb()*solver_container[FLOW_SOL]->node[iPoint]->GetDelta_Time());
634  Jacobian.AddVal2Diag(iPoint, Delta);
635 
636  /*--- Right hand side of the system (-Residual) and initial guess (x = 0) ---*/
637 
638  for (iVar = 0; iVar < nVar; iVar++) {
639  total_index = iPoint*nVar+iVar;
640  LinSysRes[total_index] = - LinSysRes[total_index];
641  LinSysSol[total_index] = 0.0;
642  AddRes_RMS(iVar, LinSysRes[total_index]*LinSysRes[total_index]);
643  AddRes_Max(iVar, fabs(LinSysRes[total_index]), geometry->node[iPoint]->GetGlobalIndex(), geometry->node[iPoint]->GetCoord());
644  }
645  }
646 
647  /*--- Initialize residual and solution at the ghost points ---*/
648 
649  for (iPoint = nPointDomain; iPoint < nPoint; iPoint++) {
650  for (iVar = 0; iVar < nVar; iVar++) {
651  total_index = iPoint*nVar + iVar;
652  LinSysRes[total_index] = 0.0;
653  LinSysSol[total_index] = 0.0;
654  }
655  }
656 
657  /*--- Solve or smooth the linear system ---*/
658 
659  CSysSolve system;
660  system.Solve(Jacobian, LinSysRes, LinSysSol, geometry, config);
661 
662  /*--- Update solution (system written in terms of increments) ---*/
663 
664  if (!adjoint) {
665 
666  /*--- Update and clip trubulent solution ---*/
667 
668  switch (config->GetKind_Turb_Model()) {
669 
670  case SA:
671 
672  for (iPoint = 0; iPoint < nPointDomain; iPoint++) {
673  node[iPoint]->AddClippedSolution(0, config->GetRelaxation_Factor_Turb()*LinSysSol[iPoint], lowerlimit[0], upperlimit[0]);
674  }
675 
676  break;
677 
678  case SA_NEG:
679 
680  for (iPoint = 0; iPoint < nPointDomain; iPoint++) {
681  node[iPoint]->AddSolution(0, config->GetRelaxation_Factor_Turb()*LinSysSol[iPoint]);
682  }
683 
684  break;
685 
686  case SST:
687 
688  for (iPoint = 0; iPoint < nPointDomain; iPoint++) {
689 
690  if (compressible) {
691  density_old = solver_container[FLOW_SOL]->node[iPoint]->GetSolution_Old(0);
692  density = solver_container[FLOW_SOL]->node[iPoint]->GetDensity();
693  }
694  if (incompressible || freesurface) {
695  density_old = solver_container[FLOW_SOL]->node[iPoint]->GetDensityInc();
696  density = solver_container[FLOW_SOL]->node[iPoint]->GetDensityInc();
697  }
698 
699  for (iVar = 0; iVar < nVar; iVar++) {
700  node[iPoint]->AddConservativeSolution(iVar, config->GetRelaxation_Factor_Turb()*LinSysSol[iPoint*nVar+iVar], density, density_old, lowerlimit[iVar], upperlimit[iVar]);
701  }
702 
703  }
704 
705  break;
706 
707  }
708  }
709 
710 
711  /*--- MPI solution ---*/
712 
713  Set_MPI_Solution(geometry, config);
714 
715  /*--- Compute the root mean square residual ---*/
716 
717  SetResidual_RMS(geometry, config);
718

721 void CTurbSolver::SetResidual_DualTime(CGeometry *geometry, CSolver **solver_container, CConfig *config,
722  unsigned short iRKStep, unsigned short iMesh, unsigned short RunTime_EqSystem) {
723 
724  /*--- Local variables ---*/
725 
726  unsigned short iVar, jVar, iMarker, iDim;
727  unsigned long iPoint, jPoint, iEdge, iVertex;
728 
729  su2double *U_time_nM1, *U_time_n, *U_time_nP1;
730  su2double Volume_nM1, Volume_nP1, TimeStep;
731  su2double Density_nM1, Density_n, Density_nP1;
732  su2double *Normal = NULL, *GridVel_i = NULL, *GridVel_j = NULL, Residual_GCL;
733 
734  bool implicit = (config->GetKind_TimeIntScheme_Turb() == EULER_IMPLICIT);
735  bool grid_movement = config->GetGrid_Movement();
736 
737  /*--- Store the physical time step ---*/
738 
739  TimeStep = config->GetDelta_UnstTimeND();
740 
741  /*--- Compute the dual time-stepping source term for static meshes ---*/
742 
743  if (!grid_movement) {
744 
745  /*--- Loop over all nodes (excluding halos) ---*/
746 
747  for (iPoint = 0; iPoint < nPointDomain; iPoint++) {
748 
749  /*--- Retrieve the solution at time levels n-1, n, and n+1. Note that
750  we are currently iterating on U^n+1 and that U^n & U^n-1 are fixed,
751  previous solutions that are stored in memory. ---*/
752 
753  U_time_nM1 = node[iPoint]->GetSolution_time_n1();
754  U_time_n = node[iPoint]->GetSolution_time_n();
755  U_time_nP1 = node[iPoint]->GetSolution();
756 
757  /*--- CV volume at time n+1. As we are on a static mesh, the volume
758  of the CV will remained fixed for all time steps. ---*/
759 
760  Volume_nP1 = geometry->node[iPoint]->GetVolume();
761 
762  /*--- Compute the dual time-stepping source term based on the chosen
763  time discretization scheme (1st- or 2nd-order).---*/
764 
765  if (config->GetKind_Turb_Model() == SST) {
766 
767  /*--- If this is the SST model, we need to multiply by the density
768  in order to get the conservative variables ---*/
769  Density_nM1 = solver_container[FLOW_SOL]->node[iPoint]->GetSolution_time_n1()[0];
770  Density_n = solver_container[FLOW_SOL]->node[iPoint]->GetSolution_time_n()[0];
771  Density_nP1 = solver_container[FLOW_SOL]->node[iPoint]->GetSolution()[0];
772 
773  for (iVar = 0; iVar < nVar; iVar++) {
774  if (config->GetUnsteady_Simulation() == DT_STEPPING_1ST)
775  Residual[iVar] = ( Density_nP1*U_time_nP1[iVar] - Density_n*U_time_n[iVar])*Volume_nP1 / TimeStep;
776  if (config->GetUnsteady_Simulation() == DT_STEPPING_2ND)
777  Residual[iVar] = ( 3.0*Density_nP1*U_time_nP1[iVar] - 4.0*Density_n*U_time_n[iVar]
778  +1.0*Density_nM1*U_time_nM1[iVar])*Volume_nP1 / (2.0*TimeStep);
779  }
780 
781  } else {
782 
783  for (iVar = 0; iVar < nVar; iVar++) {
784  if (config->GetUnsteady_Simulation() == DT_STEPPING_1ST)
785  Residual[iVar] = (U_time_nP1[iVar] - U_time_n[iVar])*Volume_nP1 / TimeStep;
786  if (config->GetUnsteady_Simulation() == DT_STEPPING_2ND)
787  Residual[iVar] = ( 3.0*U_time_nP1[iVar] - 4.0*U_time_n[iVar]
788  +1.0*U_time_nM1[iVar])*Volume_nP1 / (2.0*TimeStep);
789  }
790  }
791 
792  /*--- Store the residual and compute the Jacobian contribution due
793  to the dual time source term. ---*/
794 
795  LinSysRes.AddBlock(iPoint, Residual);
796  if (implicit) {
797  for (iVar = 0; iVar < nVar; iVar++) {
798  for (jVar = 0; jVar < nVar; jVar++) Jacobian_i[iVar][jVar] = 0.0;
799  if (config->GetUnsteady_Simulation() == DT_STEPPING_1ST)
800  Jacobian_i[iVar][iVar] = Volume_nP1 / TimeStep;
801  if (config->GetUnsteady_Simulation() == DT_STEPPING_2ND)
802  Jacobian_i[iVar][iVar] = (Volume_nP1*3.0)/(2.0*TimeStep);
803  }
804  Jacobian.AddBlock(iPoint, iPoint, Jacobian_i);
805  }
806  }
807 
808  } else {
809 
810  /*--- For unsteady flows on dynamic meshes (rigidly transforming or
811  dynamically deforming), the Geometric Conservation Law (GCL) should be
812  satisfied in conjunction with the ALE formulation of the governing
813  equations. The GCL prevents accuracy issues caused by grid motion, i.e.
814  a uniform free-stream should be preserved through a moving grid. First,
815  we will loop over the edges and boundaries to compute the GCL component
816  of the dual time source term that depends on grid velocities. ---*/
817 
818  for (iEdge = 0; iEdge < geometry->GetnEdge(); iEdge++) {
819 
820  /*--- Get indices for nodes i & j plus the face normal ---*/
821 
822  iPoint = geometry->edge[iEdge]->GetNode(0);
823  jPoint = geometry->edge[iEdge]->GetNode(1);
824  Normal = geometry->edge[iEdge]->GetNormal();
825 
826  /*--- Grid velocities stored at nodes i & j ---*/
827 
828  GridVel_i = geometry->node[iPoint]->GetGridVel();
829  GridVel_j = geometry->node[jPoint]->GetGridVel();
830 
831  /*--- Compute the GCL term by averaging the grid velocities at the
832  edge mid-point and dotting with the face normal. ---*/
833 
834  Residual_GCL = 0.0;
835  for (iDim = 0; iDim < nDim; iDim++)
836  Residual_GCL += 0.5*(GridVel_i[iDim]+GridVel_j[iDim])*Normal[iDim];
837 
838  /*--- Compute the GCL component of the source term for node i ---*/
839 
840  U_time_n = node[iPoint]->GetSolution_time_n();
841 
842  /*--- Multiply by density at node i for the SST model ---*/
843 
844  if (config->GetKind_Turb_Model() == SST) {
845  Density_n = solver_container[FLOW_SOL]->node[iPoint]->GetSolution_time_n()[0];
846  for (iVar = 0; iVar < nVar; iVar++)
847  Residual[iVar] = Density_n*U_time_n[iVar]*Residual_GCL;
848  } else {
849  for (iVar = 0; iVar < nVar; iVar++)
850  Residual[iVar] = U_time_n[iVar]*Residual_GCL;
851  }
852  LinSysRes.AddBlock(iPoint, Residual);
853 
854  /*--- Compute the GCL component of the source term for node j ---*/
855 
856  U_time_n = node[jPoint]->GetSolution_time_n();
857 
858  /*--- Multiply by density at node j for the SST model ---*/
859 
860  if (config->GetKind_Turb_Model() == SST) {
861  Density_n = solver_container[FLOW_SOL]->node[jPoint]->GetSolution_time_n()[0];
862  for (iVar = 0; iVar < nVar; iVar++)
863  Residual[iVar] = Density_n*U_time_n[iVar]*Residual_GCL;
864  } else {
865  for (iVar = 0; iVar < nVar; iVar++)
866  Residual[iVar] = U_time_n[iVar]*Residual_GCL;
867  }
868  LinSysRes.SubtractBlock(jPoint, Residual);
869 
870  }
871 
872  /*--- Loop over the boundary edges ---*/
873 
874  for (iMarker = 0; iMarker < geometry->GetnMarker(); iMarker++) {
875  for (iVertex = 0; iVertex < geometry->GetnVertex(iMarker); iVertex++) {
876 
877  /*--- Get the index for node i plus the boundary face normal ---*/
878 
879  iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
880  Normal = geometry->vertex[iMarker][iVertex]->GetNormal();
881 
882  /*--- Grid velocities stored at boundary node i ---*/
883 
884  GridVel_i = geometry->node[iPoint]->GetGridVel();
885 
886  /*--- Compute the GCL term by dotting the grid velocity with the face
887  normal. The normal is negated to match the boundary convention. ---*/
888 
889  Residual_GCL = 0.0;
890  for (iDim = 0; iDim < nDim; iDim++)
891  Residual_GCL -= 0.5*(GridVel_i[iDim]+GridVel_i[iDim])*Normal[iDim];
892 
893  /*--- Compute the GCL component of the source term for node i ---*/
894 
895  U_time_n = node[iPoint]->GetSolution_time_n();
896 
897  /*--- Multiply by density at node i for the SST model ---*/
898 
899  if (config->GetKind_Turb_Model() == SST) {
900  Density_n = solver_container[FLOW_SOL]->node[iPoint]->GetSolution_time_n()[0];
901  for (iVar = 0; iVar < nVar; iVar++)
902  Residual[iVar] = Density_n*U_time_n[iVar]*Residual_GCL;
903  } else {
904  for (iVar = 0; iVar < nVar; iVar++)
905  Residual[iVar] = U_time_n[iVar]*Residual_GCL;
906  }
907  LinSysRes.AddBlock(iPoint, Residual);
908  }
909  }
910 
911  /*--- Loop over all nodes (excluding halos) to compute the remainder
912  of the dual time-stepping source term. ---*/
913 
914  for (iPoint = 0; iPoint < nPointDomain; iPoint++) {
915 
916  /*--- Retrieve the solution at time levels n-1, n, and n+1. Note that
917  we are currently iterating on U^n+1 and that U^n & U^n-1 are fixed,
918  previous solutions that are stored in memory. ---*/
919 
920  U_time_nM1 = node[iPoint]->GetSolution_time_n1();
921  U_time_n = node[iPoint]->GetSolution_time_n();
922  U_time_nP1 = node[iPoint]->GetSolution();
923 
924  /*--- CV volume at time n-1 and n+1. In the case of dynamically deforming
925  grids, the volumes will change. On rigidly transforming grids, the
926  volumes will remain constant. ---*/
927 
928  Volume_nM1 = geometry->node[iPoint]->GetVolume_nM1();
929  Volume_nP1 = geometry->node[iPoint]->GetVolume();
930 
931  /*--- Compute the dual time-stepping source residual. Due to the
932  introduction of the GCL term above, the remainder of the source residual
933  due to the time discretization has a new form.---*/
934 
935  if (config->GetKind_Turb_Model() == SST) {
936 
937  /*--- If this is the SST model, we need to multiply by the density
938  in order to get the conservative variables ---*/
939  Density_nM1 = solver_container[FLOW_SOL]->node[iPoint]->GetSolution_time_n1()[0];
940  Density_n = solver_container[FLOW_SOL]->node[iPoint]->GetSolution_time_n()[0];
941  Density_nP1 = solver_container[FLOW_SOL]->node[iPoint]->GetSolution()[0];
942 
943  for (iVar = 0; iVar < nVar; iVar++) {
944  if (config->GetUnsteady_Simulation() == DT_STEPPING_1ST)
945  Residual[iVar] = (Density_nP1*U_time_nP1[iVar] - Density_n*U_time_n[iVar])*(Volume_nP1/TimeStep);
946  if (config->GetUnsteady_Simulation() == DT_STEPPING_2ND)
947  Residual[iVar] = (Density_nP1*U_time_nP1[iVar] - Density_n*U_time_n[iVar])*(3.0*Volume_nP1/(2.0*TimeStep))
948  + (Density_nM1*U_time_nM1[iVar] - Density_n*U_time_n[iVar])*(Volume_nM1/(2.0*TimeStep));
949  }
950 
951  } else {
952 
953  for (iVar = 0; iVar < nVar; iVar++) {
954  if (config->GetUnsteady_Simulation() == DT_STEPPING_1ST)
955  Residual[iVar] = (U_time_nP1[iVar] - U_time_n[iVar])*(Volume_nP1/TimeStep);
956  if (config->GetUnsteady_Simulation() == DT_STEPPING_2ND)
957  Residual[iVar] = (U_time_nP1[iVar] - U_time_n[iVar])*(3.0*Volume_nP1/(2.0*TimeStep))
958  + (U_time_nM1[iVar] - U_time_n[iVar])*(Volume_nM1/(2.0*TimeStep));
959  }
960  }
961 
962  /*--- Store the residual and compute the Jacobian contribution due
963  to the dual time source term. ---*/
964 
965  LinSysRes.AddBlock(iPoint, Residual);
966  if (implicit) {
967  for (iVar = 0; iVar < nVar; iVar++) {
968  for (jVar = 0; jVar < nVar; jVar++) Jacobian_i[iVar][jVar] = 0.0;
969  if (config->GetUnsteady_Simulation() == DT_STEPPING_1ST)
970  Jacobian_i[iVar][iVar] = Volume_nP1/TimeStep;
971  if (config->GetUnsteady_Simulation() == DT_STEPPING_2ND)
972  Jacobian_i[iVar][iVar] = (3.0*Volume_nP1)/(2.0*TimeStep);
973  }
974  Jacobian.AddBlock(iPoint, iPoint, Jacobian_i);
975  }
976  }
